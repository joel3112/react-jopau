<SBTitle>useTheme</SBTitle>

<SBDescription>
  Get the current theme config and dark mode. For correct functionality, it must be used inside a
  ThemeProvider.
</SBDescription>

```tsx dark
import { useTheme } from '@react-jopau/hooks';
```

<SBSubTitle>Examples</SBSubTitle>

```tsx dark
const { darkMode, onToggle } = useTheme();

console.log(darkMode); // true | false

return <button onClick={onToggle}>Toggle theme</button>;
```
